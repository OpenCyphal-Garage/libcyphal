41 template <std::size_t NumBlocksParam, std::uint8_t BlockSizeParam>
42 class LIBUAVCAN_EXPORT StaticMemoryPool final
43 {
52  union Block
53  {
59  std::uint8_t data[BlockSizeParam];
60 
66  Block* next;
67 
74  std::max_align_t alignment_type;
75  };
76 
81  typename std::aligned_storage<sizeof(Block), alignof(Block)>::type storage_[NumBlocksParam];
82 
89  std::atomic<Block*> free_list_;
90 
91  StaticMemoryPool() noexcept
92  : storage_()
93  , free_list_()
94  {
95  free_list_.store(reinterpret_cast<Block*>(&storage_));
96  static_assert(NumBlocksParam > 0, "Cannot create a PoolAllocator with less than 1 block.");
97 
98  Block* list = free_list_.load();
99  Block* next = &list[NumBlocksParam - 1];
100  next->next = nullptr;
101  for (std::size_t i = NumBlocksParam - 1; i > 0; --i)
102  {
103  Block* previous = &list[i - 1];
104  previous->next = next;
105  next = previous;
106  }
107  }
108

110  // | RULE OF SIX
111  // +------------------------------------------------------------------+
112  ~StaticMemoryPool() = default;
113

118  StaticMemoryPool& operator=(const StaticMemoryPool&&) = delete;

139  void* allocate(std::size_t size)
140  {
141  Block* previous = nullptr;
142  if (size <= BlockSizeParam)
143  {
144  Block* next = nullptr;
145  do
146  {
147  previous = free_list_.load();
148  if (previous == nullptr)
149  {
150 #if LIBUAVCAN_ENABLE_EXCEPTIONS
151  std::bad_alloc exception;
152  throw exception;
153 #endif
154  break;
155  }
156  next = previous->next;
157  } while (!std::atomic_compare_exchange_weak(&free_list_, &previous, next));
158  }
159  else
160  {
161 #if LIBUAVCAN_ENABLE_EXCEPTIONS
162  std::bad_alloc exception;
163  throw exception;
164 #endif
165  }
166 
167  return previous;
168  }
169 
170  void deallocate(void* ptr)
171  {
172  if (ptr != nullptr)
173  {
174  Block* reclaimed_block = reinterpret_cast<Block*>(ptr);
175  reclaimed_block->next = std::atomic_exchange(&free_list_, reclaimed_block);
176  }
177  }
178 };
